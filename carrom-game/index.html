(() => { const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d'); const ui = { p1: document.getElementById('p1Pill'), p2: document.getElementById('p2Pill'), p3: document.getElementById('p3Pill'), p4: document.getElementById('p4Pill'), turn: document.getElementById('turnPill'), turnText: document.getElementById('turnText'), phaseText: document.getElementById('phaseText'), queenText: document.getElementById('queenText'), shotText: document.getElementById('shotText'), powerText: document.getElementById('powerText'), meterFill: document.getElementById('meterFill'), msg: document.getElementById('msg'), mode: document.getElementById('mode'), theme: document.getElementById('theme') }; loop(); })();
